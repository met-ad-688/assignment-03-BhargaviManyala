2 Casting salary and experience columns

2.1 Computing medians and Imputing missing salaries

from pyspark.sql.functions import col

df = df.withColumn("SALARY", col("SALARY").cast("float")) \
       .withColumn("SALARY_FROM", col("SALARY_FROM").cast("float")) \
       .withColumn("SALARY_TO", col("SALARY_TO").cast("float")) \
       .withColumn("MIN_YEARS_EXPERIENCE", col("MIN_YEARS_EXPERIENCE").cast("float")) \
       .withColumn("MAX_YEARS_EXPERIENCE", col("MAX_YEARS_EXPERIENCE").cast("float"))

# Computing medians for salary columns
def compute_median(sdf, col_name):
    return sdf.approxQuantile(col_name, [0.5], 0.01)[0]

median_from = compute_median(df, "SALARY_FROM")
median_to   = compute_median(df, "SALARY_TO")
median_salary = compute_median(df, "SALARY")

print("Medians:", median_from, median_to, median_salary)

# Imputing missing salaries, but not experience
df = df.fillna({
    "SALARY_FROM": median_from,
    "SALARY_TO": median_to
})

# Computing average salary
df = df.withColumn(
    "Average_Salary", (col("SALARY_FROM") + col("SALARY_TO")) / 2
)

SELECT
    LOT_OCCUPATION_NAME AS Occupation_Name,
    PERCENTILE(SALARY, 0.5) AS Median_Salary,
    COUNT(*) AS Job_Postings
FROM job_postings
GROUP BY LOT_OCCUPATION_NAME
ORDER BY Job_Postings DESC
LIMIT 10

5 Salary by Education Level

from pyspark.sql.functions import col, when

df = df.withColumn(
    "EDU_GROUP",
    when(col("EDUCATION_LEVELS_NAME").rlike("(?i)Associate|GED|No Education Listed|Highschool"), "Associate or Lower")
    .when(col("EDUCATION_LEVELS_NAME").rlike("(?i)Bachelor"), "Bachelor")
    .when(col("EDUCATION_LEVELS_NAME").rlike("(?i)Master"), "Masters")
    .when(col("EDUCATION_LEVELS_NAME").rlike("(?i)PhD|Doctorate|professional degree"), "PhD")
    .otherwise(None)
)

df = df.withColumn("MAX_YEARS_EXPERIENCE", col("MAX_YEARS_EXPERIENCE").cast("float"))
df = df.withColumn("Average_Salary", col("Average_Salary").cast("float"))

df = df.filter(
    col("MAX_YEARS_EXPERIENCE").isNotNull() &
    (col("MAX_YEARS_EXPERIENCE") > 0) &
    col("Average_Salary").isNotNull() &
    (col("Average_Salary") > 0)
)

df_selected = df.select(

    col("EDU_GROUP"),

6 Salary by Remote Work Type

from pyspark.sql.functions import col, when, lower, trim

df = df.withColumn(
    "REMOTE_GROUP",
    when(lower(trim(col("REMOTE_TYPE_NAME"))).rlike("hyb|mix|partial|split|combo"), "Hybrid")
    .when(lower(trim(col("REMOTE_TYPE_NAME"))).rlike("remote|wfh|home|tele"), "Remote")
